\vspace{6pt}
\noindent{\textbf{Section A -- Very Short Answer Questions (10 x 2 = 20 marks)}}
\begin{enumerate}[left=0pt]
	\item Simplify: $\dfrac{x^2-9}{x-3}$ for $x\neq3$. \hfill (2)
	\item If $2x+3=11$, find $x$. \hfill (2)
	\item Find the value of $\sin 30^{\circ} + \cos 60^{\circ}$. \hfill (2)
	\item Write the slope-intercept form of the line with slope $2$ and $y$-intercept $-3$. \hfill (2)
	\item If the mean of five numbers is $12$, find their sum. \hfill (2)
	\item Expand: $(a+b)^2 - (a-b)^2$. \hfill (2)
	\item Convert $0.75$ into a fraction in simplest form. \hfill (2)
	\item Find the distance between points $(2,3)$ and $(5,7)$. \hfill (2)
	\item If the probability of an event is $\dfrac{1}{4}$, what is the probability of its complement? \hfill (2)
	\item Evaluate: $\log_{10}1000$. \hfill (2)
\end{enumerate}

\vspace{6pt}
\noindent{\textbf{Section B -- Short Answer Questions (5 x 6 = 30 marks)}}
\begin{enumerate}[left=0pt]
	\setcounter{enumi}{10}
	\item Solve the quadratic equation $x^2-5x+6=0$. Also verify by factorization. \hfill (6)
	\item A triangle has vertices at $A(0,0)$, $B(4,0)$ and $C(4,3)$. Find (a) the area of triangle ABC, (b) the equation of line AC. \hfill (6)
	\item The mean of a data set is $20$. When a new observation $40$ is added the mean becomes $21$. How many observations were there originally? Show working. \hfill (6)
	\item In right triangle PQR with right angle at Q, $\tan P = \dfrac{3}{4}$ and hypotenuse PR = 10. Find QR and PQ. \hfill (6)
	\item Factorize and solve: $3x^2-12=0$. \hfill (6)
\end{enumerate}

\vspace{6pt}
\noindent{\textbf{Section C -- Long Answer Questions (3 x 10 = 30 marks)}}
\begin{enumerate}[left=0pt]
	\setcounter{enumi}{15}
	\item Prove that the sum of the interior angles of a triangle is $180^{\circ}$. Provide a clear diagram and reasoning. \hfill (10)
	\item The quadratic polynomial $p(x)=x^2-(k+2)x+3k$ has equal roots. Find $k$ and the roots. \hfill (10)
	\item A circle has centre $O$ and radius 5 cm. A chord AB is at a distance 3 cm from the centre. Find the length of chord AB. Show your steps. \hfill (10)
\end{enumerate}

\vspace{6pt}
\noindent{\textbf{Section D -- Higher Order / Application Problems (2 x 10 = 20 marks)}}
\begin{enumerate}[left=0pt]
	\setcounter{enumi}{18}
	\item A shopkeeper offers a discount of 15% on the marked price. If the marked price is increased by 10% and then the shopkeeper gives the 15% discount, determine whether the net price is higher or lower than the original marked price and by what percent. Show calculations. \hfill (10)
	\item A fair die is rolled 120 times and the number 6 appears 26 times. Test, at the 5% level, whether this outcome is consistent with the expected frequency for a fair die. (Use approximate normal test for proportion or state reasoning.) \hfill (10)
\end{enumerate}
